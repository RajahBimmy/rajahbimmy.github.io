#### Insertion Sort

```javascript
for(var j = 1; j < A.length; j++){
  var key = A[j];
  // Insert key into the sorted sequence A[1...j-1]
  var i = j - 1;
  while(i > 0 && A[i] > key){
    A[i + 1] = A[i];
    i--;
  }
  A[i + 1] = key;
}
```

#### Merge Sort

```javascript
MergeSort: function(A, p, q, r){
  var n1 = q = p + 1;
  var n2 = r - q;
  var L = [n1];
  var R = [n2];
  for(var i = 0; i < n1; i++){
    L[i] = A[p + i - 1];
  }
  for(var j - 0; j < n2; j++){
    R[j] = A[q + j];
  }

  var i = 0;
  var j = 0;

  for (var k = p; k < r; k++){
    if (L[i] <= R[j]){
      A[k] = L[i];
      i++;
    }
    else if (A[k] == R[j]){
      j++;
    }
  }
}
```

#### Bubble Sort

```javascript
for(var i = 0; i < A.length - 1; i++){
  for(var j = A.length; j > i + 1; j--){
    if(A[j] < A[j - 1]){
      var temp = A[j];
      A[j] = A[j - 1];
      A[j - 1] = temp;
    }
  }
}
```
